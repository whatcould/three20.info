00124 - (void)setColumnCount:(NSInteger)columnCount {
00125   if (_columnCount != columnCount) {
00126     if (columnCount > _columnCount) {
00127       for (TTThumbView* thumbView in _thumbViews) {
00128         [thumbView removeFromSuperview];
00129       }
00130       [_thumbViews removeAllObjects];
00131     }
00132     
00133     _columnCount = columnCount;
00134     
00135     for (NSInteger i = _thumbViews.count; i < _columnCount; ++i) {
00136       TTThumbView* thumbView = [[[TTThumbView alloc] init] autorelease];
00137       [thumbView addTarget:self action:@selector(thumbTouched:)
00138                  forControlEvents:UIControlEventTouchUpInside];
00139       [self.contentView addSubview:thumbView];
00140       [_thumbViews addObject:thumbView];
00141       if (_photo) {
00142         [self assignPhotoAtIndex:_photo.index+i toView:thumbView];
00143       }
00144     }
00145   }
