00118 - (void)setContentOffset:(CGPoint)point {
00119   // UITableView (and UIScrollView) are really stupid about resetting the content offset
00120   // when the table view itself is resized.  There are times when I scroll to a point and then
00121   // disable scrolling, and I don't want the table view scrolling somewhere else just because
00122   // it was resized.  
00123   if (self.scrollEnabled) {
00124     if (!(_contentOrigin && self.contentOffset.y == _contentOrigin && point.y == 0)) {
00125       [super setContentOffset:point];
00126     }
00127   }
