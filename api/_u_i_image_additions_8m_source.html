00027 - (void)addRoundedRectToPath:(CGContextRef)context rect:(CGRect)rect radius:(float)radius {
00028   CGContextBeginPath(context);
00029   CGContextSaveGState(context);
00030 
00031   if (radius == 0) {
00032     CGContextTranslateCTM(context, CGRectGetMinX(rect), CGRectGetMinY(rect));
00033     CGContextAddRect(context, rect);
00034   } else {
00035     CGContextTranslateCTM(context, CGRectGetMinX(rect), CGRectGetMinY(rect));
00036     CGContextScaleCTM(context, radius, radius);
00037     float fw = CGRectGetWidth(rect) / radius;
00038     float fh = CGRectGetHeight(rect) / radius;
00039     
00040     CGContextMoveToPoint(context, fw, fh/2);
00041     CGContextAddArcToPoint(context, fw, fh, fw/2, fh, 1);
00042     CGContextAddArcToPoint(context, 0, fh, 0, fh/2, 1);
00043     CGContextAddArcToPoint(context, 0, 0, fw/2, 0, 1);
00044     CGContextAddArcToPoint(context, fw, 0, fw, fh/2, 1);
00045   }
00046 
00047   CGContextClosePath(context);
00048   CGContextRestoreGState(context);
00049 }

00054 - (UIImage*)transformWidth:(CGFloat)width height:(CGFloat)height rotate:(BOOL)rotate {
00055   CGFloat destW = width;
00056   CGFloat destH = height;
00057   CGFloat sourceW = width;
00058   CGFloat sourceH = height;
00059   if (rotate) {
00060     if (self.imageOrientation == UIImageOrientationRight
00061         || self.imageOrientation == UIImageOrientationLeft) {
00062       sourceW = height;
00063       sourceH = width;
00064     }
00065   }
00066   
00067   CGImageRef imageRef = self.CGImage;
00068   CGContextRef bitmap = CGBitmapContextCreate(NULL, destW, destH,
00069     CGImageGetBitsPerComponent(imageRef), CGImageGetBytesPerRow(imageRef), CGImageGetColorSpace(imageRef),
00070     CGImageGetBitmapInfo(imageRef));
00071 
00072   if (rotate) {
00073     if (self.imageOrientation == UIImageOrientationDown) {
00074       CGContextTranslateCTM(bitmap, sourceW, sourceH);
00075       CGContextRotateCTM(bitmap, 180 * (M_PI/180));
00076     } else if (self.imageOrientation == UIImageOrientationLeft) {
00077       CGContextTranslateCTM(bitmap, sourceH, 0);
00078       CGContextRotateCTM(bitmap, 90 * (M_PI/180));
00079     } else if (self.imageOrientation == UIImageOrientationRight) {
00080       CGContextTranslateCTM(bitmap, 0, sourceW);
00081       CGContextRotateCTM(bitmap, -90 * (M_PI/180));
00082     }
00083   }
00084 
00085   CGContextDrawImage(bitmap, CGRectMake(0,0,sourceW,sourceH), imageRef);
00086 
00087   CGImageRef ref = CGBitmapContextCreateImage(bitmap);
00088   UIImage* result = [UIImage imageWithCGImage:ref];
00089   CGContextRelease(bitmap);
00090   CGImageRelease(ref);
00091
