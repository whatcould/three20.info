00064 - (void)setHighlightedFrame:(TTStyledBoxFrame*)frame{
00065   if (frame != _highlightedFrame) {
00066     TTTableView* tableView = (TTTableView*)[self ancestorOrSelfWithClass:[TTTableView class]];
00067     
00068     TTStyledBoxFrame* affectFrame = frame ? frame : _highlightedFrame;
00069     NSString* className = affectFrame.element.className;
00070     if (!className && [affectFrame.element isKindOfClass:[TTStyledLinkNode class]]) {
00071       className = @"linkText:";
00072     }
00073     
00074     if (className && [className rangeOfString:@":"].location != NSNotFound) {
00075       if (frame) {
00076         TTStyle* style = [TTSTYLESHEET styleWithSelector:className
00077                                        forState:UIControlStateHighlighted];
00078         [self setStyle:style forFrame:frame];
00079         
00080         [_highlightedFrame release];
00081         _highlightedFrame = [frame retain];
00082         [_highlightedNode release];
00083         _highlightedNode = [frame.element retain];
00084         tableView.highlightedLabel = self;
00085       } else {
00086         TTStyle* style = [TTSTYLESHEET styleWithSelector:className forState:UIControlStateNormal];
00087         [self setStyle:style forFrame:_highlightedFrame];
00088 
00089         TT_RELEASE_SAFELY(_highlightedFrame);
00090         TT_RELEASE_SAFELY(_highlightedNode);
00091         tableView.highlightedLabel = nil;
00092       }
00093 
00094       [self setNeedsDisplay];
00095     }
00096   }

00099 - (NSString*)combineTextFromFrame:(TTStyledTextFrame*)fromFrame toFrame:(TTStyledTextFrame*)toFrame {
00100   NSMutableArray* strings = [NSMutableArray array];
00101   for (TTStyledTextFrame* frame = fromFrame; frame && frame != toFrame;
00102        frame = (TTStyledTextFrame*)frame.nextFrame) {
00103     [strings addObject:frame.text];

00131 - (void)addAccessibilityElementsForNode:(TTStyledNode*)node {
00132   if ([node isKindOfClass:[TTStyledLinkNode class]]) {
00133     UIAccessibilityElement* acc = [[[UIAccessibilityElement alloc]
00134                                   initWithAccessibilityContainer:self] autorelease];
00135     TTStyledFrame* frame = [_text getFrameForNode:node];
00136     acc.accessibilityFrame = CGRectOffset(frame.bounds, self.screenViewX, self.screenViewY);
00137     acc.accessibilityTraits = UIAccessibilityTraitLink;
00138     acc.accessibilityLabel = [node outerText];
00139     [_accessibilityElements addObject:acc];
00140   } else if ([node isKindOfClass:[TTStyledTextNode class]]) {
00141     TTStyledTextFrame* startFrame = (TTStyledTextFrame*)[_text getFrameForNode:node];
00142     UIEdgeInsets edges = [self edgesForRect:startFrame.bounds];
00143   
00144     TTStyledTextFrame* frame = (TTStyledTextFrame*)startFrame.nextFrame;
00145     for (; [frame isKindOfClass:[TTStyledTextFrame class]]; frame = (TTStyledTextFrame*)frame.nextFrame) {
00146       if (frame.bounds.origin.x < edges.left) {
00147         [self addAccessibilityElementFromFrame:startFrame toFrame:frame withEdges:edges];
00148         edges = [self edgesForRect:frame.bounds];
00149         startFrame = frame;
00150       } else {
00151         if (frame.bounds.origin.x+frame.bounds.size.width > edges.right) {
00152           edges.right = frame.bounds.origin.x+frame.bounds.size.width;
00153         }
00154         if (frame.bounds.origin.y+frame.bounds.size.height > edges.bottom) {
00155           edges.bottom = frame.bounds.origin.y+frame.bounds.size.height;
00156         }
00157       }
00158     }
00159     
00160     if (frame != startFrame) {
00161       [self addAccessibilityElementFromFrame:startFrame toFrame:frame withEdges:edges];
00162     }
00163   } else if ([node isKindOfClass:[TTStyledElement class]]) {
00164     TTStyledElement* element = (TTStyledElement*)node;
00165     for (TTStyledNode* child = element.firstChild; child; child = child.nextSibling) {
00166       [self addAccessibilityElementsForNode:child];
00167     }
00168   }
00169 }
