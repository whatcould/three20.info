00059   CGFloat x = _padding, y = _padding;

00062   for (UIView* subview in subviews) {
00063     if (x + subview.width > maxWidth) {
00064       x = _padding;

00107   CGFloat innerWidth = (view.width - _padding*2);
00108   CGFloat width = ceil(innerWidth / _columnCount);
00109   CGFloat rowHeight = 0;

00116       x = _padding;
