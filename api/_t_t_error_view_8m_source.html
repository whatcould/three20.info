00088                       + kVPadding1 + kVPadding2;
00089   BOOL canShowImage = _imageView.image && self.height > maxHeight;
00090   
00091   CGFloat totalHeight = 0;
00092 
00093   if (canShowImage) {
00094     totalHeight += _imageView.height;
00095   }
00096   if (_titleView.text.length) {
00097     totalHeight += (totalHeight ? kVPadding1 : 0) + _titleView.height;

00099   if (_subtitleView.text.length) {
00100     totalHeight += (totalHeight ? kVPadding2 : 0) + _subtitleView.height;
00101   }
00102   
00103   CGFloat top = floor(self.height/2 - totalHeight/2);
00104   
00105   if (canShowImage) {
00106     _imageView.origin = CGPointMake(floor(self.width/2 - _imageView.width/2), top);
00107     _imageView.hidden = NO;
00108     top += _imageView.height + kVPadding1;
00109   } else {
00110     _imageView.hidden = YES;
00111   }
00112   if (_titleView.text.length) {
00113     _titleView.origin = CGPointMake(floor(self.width/2 - _titleView.width/2), top);

00117     _subtitleView.origin = CGPointMake(floor(self.width/2 - _subtitleView.width/2), top);
