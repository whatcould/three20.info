00090 CGAffineTransform TTRotateTransformForOrientation(UIInterfaceOrientation orientation) {
00091   if (orientation == UIInterfaceOrientationLandscapeLeft) {
00092     return CGAffineTransformMakeRotation(M_PI*1.5);
00093   } else if (orientation == UIInterfaceOrientationLandscapeRight) {
00094     return CGAffineTransformMakeRotation(M_PI/2);
00095   } else if (orientation == UIInterfaceOrientationPortraitUpsideDown) {
00096     return CGAffineTransformMakeRotation(-M_PI);
00097   } else {
00098     return CGAffineTransformIdentity;
00099   }
00100 }
