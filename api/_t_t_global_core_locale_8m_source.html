00043 NSString* TTDescriptionForError(NSError* error) {
00044   TTDINFO(@"ERROR %@", error);
00045   if ([error.domain isEqualToString:NSURLErrorDomain]) {
00046     if (error.code == NSURLErrorTimedOut) {
00047       return TTLocalizedString(@"Connection Timed Out", @"");
00048     } else if (error.code == NSURLErrorNotConnectedToInternet) {
00049       return TTLocalizedString(@"No Internet Connection", @"");
00050     } else {
00051       return TTLocalizedString(@"Connection Error", @"");

00053   }
00054   return TTLocalizedString(@"Error", @"");
