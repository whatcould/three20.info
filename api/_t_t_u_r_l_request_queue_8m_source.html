00572     if ([delegate respondsToSelector:@selector(requestDidStartLoad:)]) {
00573       [delegate requestDidStartLoad:request];

00578     NSError* error = [NSError errorWithDomain:NSURLErrorDomain code:NSURLErrorBadURL userInfo:nil];
00579     for (id<TTURLRequestDelegate> delegate in request.delegates) {

00591     // Next, see if there is an active loader for the URL and if so join that bandwagon
00592     loader = [_loaders objectForKey:request.cacheKey];

00594       [loader addRequest:request];

00602   if (_suspended || _totalLoading == kMaxConcurrentLoads) {
00603     [_loaderQueue addObject:loader];

00605     ++_totalLoading;
00606     [loader load:[NSURL URLWithString:request.URL]];
